15pub(super) fn compile_instruction(ctx: &mut Context, data: MatchData) -> Result<(), Option<String>> {
16    let mut cursor = 0usize;
17
18    // All static bitfields (compile-time constant) will be encoded into this map of (offset, bitfield)
19    let mut statics = Vec::new();
20    // All dynamic bitfields (run-time determined) will be encoded into this map of (offset, TokenStream)
21    let mut dynamics = Vec::new();
22    // Any relocations will be encoded into this list
23    let mut relocations = Vec::new();
24
25    for command in data.data.commands.iter() {
26        match *command {
27            // special commands that don't check the current arg
28            Command::A => {
29                cursor += 1;
30                continue
31            },
32            Command::C => {
33                cursor -= 1;
34                continue
35            },
36            Command::Rwidth(offset) => {
37                statics.push((offset, data.simd_full_width.unwrap_or(true) as u32));
38                continue
39            },
40
41            _ => ()
42        }
43
44        let arg = data.args.get(cursor).expect("Invalid encoding data, tried to process more arguments than given");
45
46        match *arg {
47            FlatArg::Direct { span, reg: RegKind::Static(id) } => match *command {
48                Command::R(offset) => {
49                    statics.push((offset, u32::from(id.code())));
50                },
51                Command::REven(offset) => {
52                    if id.code() & 1 != 0 {
53                        emit_error!(span, "Field only supports even registers");
54                        return Err(None);
55                    }
56                    statics.push((offset, u32::from(id.code())));
57                },
58                Command::RNoZr(offset) => {
59                    if id.code() == 31 {
60                        emit_error!(span, "Field does not support register the zr/sp register");
61                        return Err(None);
62                    }
63                    statics.push((offset, u32::from(id.code())));
64                },
65                Command::R4(offset) => {
66                    if id.code() >= 16 {
67                        emit_error!(span, "Field only supports register numbers 0-15");
68                        return Err(None);
69                    }
70                    statics.push((offset, u32::from(id.code())));
71                },
72                Command::RNext => {
73                    if let Some(FlatArg::Direct { span: _prevspan, reg: ref prevreg } ) = data.args.get(cursor - 1) {
74                        match prevreg {
75                            RegKind::Static(previd) => if id.code() != ((previd.code() + 1) % 32) {
76                                emit_error!(span, "Invalid register. This register has to be the register after the previous argument.");
77                                return Err(None);
78                            },
79                            RegKind::Dynamic(_, _) => if id != RegId::XZR {
80                                emit_error!(span, "Please use XZR here to indicate that it should be the register after the previous argument.");
81                                return Err(None);
82                            }
83                        }
84                    } else {
85                        panic!("RNext command without the previous command being a register encoder");
86                    }
87                },
88                _ => panic!("Invalid argument processor")
89            },
90            FlatArg::Direct { span, reg: RegKind::Dynamic(_, ref expr) } => match *command {
91                Command::R(offset)
92                | Command::RNoZr(offset) => {
93                    dynamics.push((offset, quote_spanned!{ span=>
94                        #expr & 0x1F
95                    }));
96                },
97                Command::REven(offset) => {
98                    dynamics.push((offset, quote_spanned!{ span=>
99                        #expr & 0x1E
100                    }));
101                },
102                Command::R4(offset) => {
103                    dynamics.push((offset, quote_spanned!{ span=>
104                        #expr & 0xF
105                    }));
106                },
107                Command::RNext => {
108                    emit_error!(span, "This register is constrained to be the register after the previous argument's register. As such, it does not support dynamic registers. Please substitute it with XZR to indicate this");
109                    return Err(None);
110                },
111                _ => panic!("Invalid argument processor")
112            },
113            FlatArg::Modifier { modifier, .. } => match *command {
114                Command::Rotates(offset) => match modifier {
115                    Modifier::LSL => statics.push((offset, 0b00)),
116                    Modifier::LSR => statics.push((offset, 0b01)),
117                    Modifier::ASR => statics.push((offset, 0b10)),
118                    Modifier::ROR => statics.push((offset, 0b11)),
119                    _ => panic!("Unexpected modifier for argument processor")
120                },
121                Command::ExtendsW(offset) => match modifier {
122                    Modifier::UXTB => statics.push((offset, 0b000)),
123                    Modifier::UXTH => statics.push((offset, 0b001)),
124                    Modifier::UXTW => statics.push((offset, 0b010)),
125                    Modifier::UXTX => statics.push((offset, 0b011)),
126                    Modifier::SXTB => statics.push((offset, 0b100)),
127                    Modifier::SXTH => statics.push((offset, 0b101)),
128                    Modifier::SXTW => statics.push((offset, 0b110)),
129                    Modifier::SXTX => statics.push((offset, 0b111)),
130                    Modifier::LSL  => statics.push((offset, 0b010)),
131                    _ => panic!("Unexpected modifier for argument processor")
132                },
133                Command::ExtendsX(offset) => match modifier {
134                    Modifier::UXTB => statics.push((offset, 0b000)),
135                    Modifier::UXTH => statics.push((offset, 0b001)),
136                    Modifier::UXTW => statics.push((offset, 0b010)),
137                    Modifier::UXTX => statics.push((offset, 0b011)),
138                    Modifier::SXTB => statics.push((offset, 0b100)),
139                    Modifier::SXTH => statics.push((offset, 0b101)),
140                    Modifier::SXTW => statics.push((offset, 0b110)),
141                    Modifier::SXTX => statics.push((offset, 0b111)),
142                    Modifier::LSL  => statics.push((offset, 0b011)),
143                    _ => panic!("Unexpected modifier for argument processor")
144                },
145                _ => panic!("Invalid argument processor")
146            },
147            FlatArg::Immediate { ref value } => match *command {
148
149                // Condition codes, literals
150                Command::Cond(offset) => {
151                    let name = as_ident(value).expect("bad command data").to_string();
152                    let bits = *COND_MAP.get(&&*name).expect("bad command data");
153                    statics.push((offset, u32::from(bits)))
154                },
155                Command::CondInv(offset) => {
156                    let name = as_ident(value).expect("bad command data").to_string();
157                    let bits = *COND_MAP.get(&&*name).expect("bad command data");
158                    statics.push((offset, u32::from(bits) ^ 1))
159                },
160                Command::LitList(offset, listname) => {
161                    let name = as_ident(value).expect("bad command data").to_string();
162                    let list = SPECIAL_IDENT_MAP.get(listname).expect("bad command data");
163                    if let Some(&bits) = list.get(&&*name) {
164                        statics.push((offset, bits));
165                    } else {
166                        emit_error!(value, "Unknown literal");
167                        return Err(None);
168                    }
169                },
170
171                // unsigned immediate encodings
172
173                Command::Ubits(offset, bitlen) => {
174                    let mask = bitmask(bitlen);
175
176                    if let Some((biased, _)) = static_range_check(value, 0, mask, 0)? {
177                        statics.push((offset, biased));
178
179                    } else {
180                        let check = dynamic_range_check_unsigned(value.span(), 0, mask, 0);
181
182                        dynamics.push((offset, quote_spanned!{ value.span()=>
183                            { let _dyn_imm: u32 = #value; #check; _dyn_imm & #mask }
184                        }));
185                    }
186                },
187                Command::Uscaled(offset, bitlen, shift) => {
188                    let mask = bitmask(bitlen);
189
190                    if let Some((biased, _)) = static_range_check(value, 0, mask, shift)? {
191                        statics.push((offset, biased));
192
193                    } else {
194                        let check = dynamic_range_check_unsigned(value.span(), 0, mask, shift);
195
196                        dynamics.push((offset, quote_spanned!{ value.span()=>
197                            { let _dyn_imm: u32 = #value; #check; (_dyn_imm >> #shift) & #mask }
198                        }));
199                    }
200                },
201                Command::Uslice(offset, bitlen, shift) => {
202                    let mask = bitmask(bitlen);
203
204                    if let Some(value) = as_unsigned_number(value) {
205                        statics.push((offset, ((value as u32) >> shift) & mask));
206
207                    } else {
208                        dynamics.push((offset, quote_spanned!{ value.span()=>
209                            { let _dyn_imm: u32 = #value; (#value >> #shift) & #mask }
210                        }));
211                    }
212                },
213                Command::Ulist(offset, options) => {
214                    if let Some(number) = as_unsigned_number(value) {
215                        if let Some(i) = options.iter().rposition(|&n| u64::from(n) == number) {
216                            statics.push((offset, i as u32));
217
218                        } else {
219                            emit_error!(value, "Invalid immediate '{}'", number);
220                            return Err(None);
221                        }
222                    } else {
223                        dynamics.push((offset, quote_spanned!{ value.span()=>
224                            {
225                                let _dyn_imm = #value;
226                                [#(#options),*].iter().rposition(|&n| n as u32 == _dyn_imm)
227                                    .unwrap_or_else(|| ::dynasmrt::aarch64::immediate_out_of_range_unsigned_32(_dyn_imm)) as u32
228                            }
229                        }));
230                    }
231                },
232                Command::Urange(offset, min, max) => {
233                    let range = u32::from(max - min);
234                    let min = u32::from(min);
235                    if let Some((biased, _)) = static_range_check(value, min as i32, range, 0)? {
236                        statics.push((offset, biased));
237
238                    } else {
239                        let check = dynamic_range_check_unsigned(value.span(), min, range, 0);
240
241                        let mask = (range + 1).next_power_of_two() - 1;
242
243                        dynamics.push((offset, quote_spanned!{ value.span()=>
244                            { let _dyn_imm: u32 = #value; #check; (_dyn_imm - #min) & #mask }
245                        }));
246                    }
247                },
248                Command::Usubone(offset, bitlen) => {
249                    let mask = bitmask(bitlen);
250
251                    if let Some((biased, _)) = static_range_check(value, 1, mask, 0)? {
252                        statics.push((offset, mask - biased));
253
254                    } else {
255                        let check = dynamic_range_check_unsigned(value.span(), 1, mask, 0);
256
257                        let top = mask + 1;
258                        dynamics.push((offset, quote_spanned!{ value.span()=>
259                            { let _dyn_imm: u32 = #value; #check; (#top - _dyn_imm) & #mask }
260                        }));
261                    }
262                },
263                Command::Usubzero(offset, bitlen) => {
264                    let mask = bitmask(bitlen);
265
266                    if let Some((biased, _)) = static_range_check(value, 0, mask, 0)? {
267                        statics.push((offset, mask - biased));
268
269                    } else {
270                        let check = dynamic_range_check_unsigned(value.span(), 0, mask, 0);
271
272                        dynamics.push((offset, quote_spanned!{ value.span()=>
273                            { let _dyn_imm: u32 = #value; #check; (#mask - _dyn_imm) & #mask }
274                        }));
275                    }
276                },
277                Command::Usubmod(offset, bitlen) => {
278                    let mask = bitmask(bitlen);
279
280                    if let Some((biased, _)) = static_range_check(value, 0, mask, 0)? {
281                        statics.push((offset, biased.wrapping_neg() & mask));
282
283                    } else {
284                        let check = dynamic_range_check_unsigned(value.span(), 0, mask, 0);
285
286                        dynamics.push((offset, quote_spanned!{ value.span()=>
287                            { let _dyn_imm: u32 = #value; #check; _dyn_imm.wrapping_neg() & #mask }
288                        }));
289                    }
290                },
291                Command::Usum(offset, bitlen) => {
292                    let mask = bitmask(bitlen);
293                    let prev_value = if let Some(FlatArg::Immediate {value: prev_value } ) = data.args.get(cursor - 1) {
294                        prev_value
295                    } else {
296                        panic!("Bad encoding data, previous argument was not an immediate");
297                    };
298
299                    let number = if let Some(prev_number) = as_unsigned_number(prev_value) {
300                        if prev_number > mask as u64 {
301                            emit_error!(prev_value, "Impossible immediate combination");
302                            return Err(None);
303                        };
304
305                        if let Some((biased, _)) = static_range_check(value, 1, mask - (prev_number as u32), 0)? {
306                            Some(biased + (prev_number as u32))
307                        } else {
308                            None
309                        }
310                    } else {
311                        None
312                    };
313
314                    if let Some(number) = number {
315                        statics.push((offset, number & mask));
316                    } else {
317                        let check = quote_spanned!{ value.span()=>
318                            if (#value - 1u32) > (#mask - #prev_value) { ::dynasmrt::aarch64::immediate_out_of_range_unsigned_32(#value); }
319                        };
320
321                        dynamics.push((offset, quote_spanned!{ value.span()=>
322                            { let _dyn_imm: u32 = #value; #check; (#prev_value + _dyn_imm - 1) & #mask }
323                        }));
324                    }
325                },
326                Command::Ufields(bitfields) => {
327                    let mask = bitmask(bitfields.len() as u8);
328
329                    if let Some((biased, _)) = static_range_check(value, 0, mask, 0)? {
330                        for (i, &field) in bitfields.iter().rev().enumerate() {
331                            statics.push((field, (biased >> i) & 1));
332                        }
333
334                    } else {
335                        let check = dynamic_range_check_unsigned(value.span(), 0, mask, 0);
336
337                        for (i, &field) in bitfields.iter().rev().enumerate() {
338                            if i == 0 {
339                                dynamics.push((field, quote_spanned!{ value.span()=>
340                                    { let _dyn_imm: u32 = #value; #check; (_dyn_imm >> #i) & 1 }
341                                }));
342
343                            } else {
344                                dynamics.push((field, quote_spanned!{ value.span()=>
345                                    (#value >> #i) & 1
346                                }));
347                            }
348                        }
349                    }
350                },
351
352                // signed immediate encoding
353
354                Command::Sbits(offset, bitlen) => {
355                    let mask = bitmask(bitlen);
356                    let half = -1i32 << (bitlen - 1);
357
358                    if let Some((_, scaled)) = static_range_check(value, half, mask, 0)? {
359                        statics.push((offset, scaled & mask));
360
361                    } else {
362                        let check = dynamic_range_check_signed(value.span(), half, mask, 0);
363
364                        dynamics.push((offset, quote_spanned!{ value.span()=>
365                            { let _dyn_imm: i32 = #value; #check; (_dyn_imm as u32) & #mask }
366                        }));
367                    }
368                },
369                Command::Sscaled(offset, bitlen, shift) => {
370                    let mask = bitmask(bitlen);
371                    let half = -1i32 << (bitlen - 1);
372
373                    if let Some((_, scaled)) = static_range_check(value, half, mask, shift)? {
374                        statics.push((offset, scaled & mask));
375
376                    } else {
377                        let check = dynamic_range_check_signed(value.span(), half, mask, shift);
378
379                        dynamics.push((offset, quote_spanned!{ value.span()=>
380                            { let _dyn_imm: i32 = #value; #check; ((_dyn_imm >> #shift) as u32) & #mask }
381                        }));
382                    }
383                },
384                Command::Sslice(offset, bitlen, shift) => {
385                    let mask = bitmask(bitlen);
386
387                    if let Some(value) = as_signed_number(value) {
388                        statics.push((offset, ((value >> shift) as u32) & mask));
389
390                    } else {
391                        dynamics.push((offset, quote_spanned!{ value.span()=>
392                            { let _dyn_imm: i32 = #value; ((_dyn_imm >> #shift) as u32) & #mask }
393                        }));
394                    }
395                },
396
397                // pure validation encodings.
398
399                Command::CUbits(bitlen) => {
400                    let mask = bitmask(bitlen);
401
402                    if static_range_check(value, 0, mask, 0)?.is_none() {
403                        let check = dynamic_range_check_unsigned(value.span(), 0, mask, 0);
404
405                        dynamics.push((0, quote_spanned! { value.span()=>
406                            { let _dyn_imm: u32 = #value; #check; 0 }
407                        }));
408                    }
409                },
410                Command::CUsum(bitlen) => {
411                    let mask = bitmask(bitlen);
412
413                    let prev_value = if let Some(FlatArg::Immediate {value: prev_value } ) = data.args.get(cursor - 1) {
414                        prev_value
415                    } else {
416                        panic!("Bad encoding data, previous argument was not an immediate");
417                    };
418
419                    let check = if let Some(prev_number) = as_unsigned_number(prev_value) {
420                        if prev_number > mask as u64 {
421                            emit_error!(prev_value, "Impossible immediate combination");
422                            return Err(None);
423                        };
424
425                        static_range_check(value, 1, mask - (prev_number as u32), 0)?
426                    } else {
427                        None
428                    };
429
430                    if check.is_none() {
431
432                        let check = quote_spanned!{ value.span()=>
433                            if (#value - 1u32) > (#mask - #prev_value) { ::dynasmrt::aarch64::immediate_out_of_range_unsigned_32(#value); }
434                        };
435
436                        dynamics.push((0, quote_spanned! { value.span()=>
437                            { #check; 0}
438                        }));
439                    }
440                },
441                Command::CSscaled(bitlen, shift) => {
442                    let mask = bitmask(bitlen);
443                    let half = -1i32 << (bitlen - 1);
444
445                    if static_range_check(value, half, mask, shift)?.is_none() {
446                        let check = dynamic_range_check_signed(value.span(), half, mask, shift);
447                        dynamics.push((0, quote_spanned! { value.span()=>
448                            { let _dyn_imm: i32 = #value; #check; 0 }
449                        }));
450                    }
451                },
452                Command::CUrange(min, max) => {
453                    let range = u32::from(max - min);
454                    let min = u32::from(min);
455
456                    if static_range_check(value, min as i32, range, 0)?.is_none() {
457                        let check = dynamic_range_check_unsigned(value.span(), min, range, 0);
458                        dynamics.push((0, quote_spanned! { value.span()=>
459                            { let _dyn_imm: u32 = #value; #check; 0 }
460                        }));
461                    }
462                },
463
464                // specials. These have some more involved code.
465                Command::Special(offset, special) => handle_special_immediates(offset, special, value, &mut statics, &mut dynamics)?,
466
467                // jump targets also accept immediates
468                Command::Offset(relocation) => match relocation {
469                     // b, bl 26 bits, dword aligned
470                    Relocation::B => {
471                        let bits = 26;
472                        let mask = bitmask(bits);
473                        let half = -1i32 << (bits - 1);
474                        if let Some((_, scaled)) = static_range_check(value, half, mask, 2)? {
475                            statics.push((0, scaled & mask));
476
477                        } else {
478                            let check = dynamic_range_check_signed(value.span(), half, mask, 2);
479
480                            dynamics.push((0, quote_spanned!{ value.span()=>
481                                { let _dyn_imm: i32 = #value; #check; ((_dyn_imm >> 2u8) as u32) & #mask }
482                            }));
483                        }
484                    },
485                    // b.cond, cbnz, cbz, ldr, ldrsw, prfm: 19 bits, dword aligned
486                    Relocation::BCOND => {
487                        let bits = 19;
488                        let mask = bitmask(bits);
489                        let half = -1i32 << (bits - 1);
490                        if let Some((_, scaled)) = static_range_check(value, half, mask, 2)? {
491                            statics.push((5, scaled & mask));
492
493                        } else {
494                            let check = dynamic_range_check_signed(value.span(), half, mask, 2);
495
496                            dynamics.push((5, quote_spanned!{ value.span()=>
497                                { let _dyn_imm: i32 = #value; #check; ((_dyn_imm >> 2u8) as u32) & #mask }
498                            }));
499                        }
500                    },
501                    // adr split 21 bit, byte aligned
502                    Relocation::ADR => {
503                        let bits = 21;
504                        let mask = bitmask(bits);
505                        let half = -1i32 << (bits - 1);
506                        if let Some((_, scaled)) = static_range_check(value, half, mask, 0)? {
507                            statics.push((5, (scaled >> 2u8) & 0x7FFFFu32));
508                            statics.push((29, scaled & 3u32));
509
510                        } else {
511                            let check = dynamic_range_check_signed(value.span(), half, mask, 0);
512                            dynamics.push((5, quote_spanned!{ value.span()=>
513                                { let _dyn_imm: i32 = #value; #check; ((_dyn_imm >> 2u8) as u32) & 0x7FFFFu32 }
514                            }));
515                            dynamics.push((29, quote_spanned!{ value.span()=>
516                                (#value as u32) & 3u32
517                            }));
518                        }
519                    },
520                    // adrp split 21 bit, 4096-byte aligned
521                    Relocation::ADRP => {
522                        let bits = 21;
523                        let mask = bitmask(bits);
524                        let half = -1i32 << (bits - 1);
525                        if let Some((_, scaled)) = static_range_check(value, half, mask, 12)? {
526                            statics.push((5, scaled & 0x7FFFF));
527                            statics.push((29, scaled & 3));
528
529                        } else {
530                            let check = dynamic_range_check_signed(value.span(), half, mask, 12);
531                            dynamics.push((5, quote_spanned!{ value.span()=>
532                                { let _dyn_imm: i32 = #value; #check; ((_dyn_imm >> 14u8) as u32) & 0x7FFFFu32 }
533                            }));
534                            dynamics.push((29, quote_spanned!{ value.span()=>
535                                ((#value >> 12u8) as u32) & 3u32
536                            }));
537                        }
538                    },
539                    // tbnz, tbz: 14 bits, dword aligned
540                    Relocation::TBZ => {
541                        let bits = 14;
542                        let mask = bitmask(bits);
543                        let half = -1i32 << (bits - 1);
544                        if let Some((_, scaled)) = static_range_check(value, half, mask, 2)? {
545                            statics.push((5, scaled & mask));
546
547                        } else {
548                            let check = dynamic_range_check_signed(value.span(), half, mask, 2);
549                            dynamics.push((5, quote_spanned!{ value.span()=>
550                                { let _dyn_imm: i32 = #value; #check; ((_dyn_imm >> 2) as u32) & #mask }
551                            }));
552                        }
553                    },
554                    Relocation::LITERAL8
555                    | Relocation::LITERAL16
556                    | Relocation::LITERAL32
557                    | Relocation::LITERAL64 => ()
558                },
559
560                _ => panic!("Invalid argument processor")
561            },
562            FlatArg::Default => match *command {
563                // Registers default to R31
564                Command::R(offset) => {
565                    statics.push((offset, 0b11111u32));
566                },
567
568                // modifiers to LSL
569                Command::Rotates(offset) => {
570                    statics.push((offset, 0b00));
571                },
572                Command::ExtendsW(offset) => {
573                    statics.push((offset, 0b010));
574                },
575                Command::ExtendsX(offset) => {
576                    statics.push((offset, 0b011));
577                },
578
579                // normal integer encodings default to 0 (i.e. not doing anything)
580                // however encoders for which 0 is not necessarily a valid value cannot match default
581                Command::Ubits(_, _) |
582                Command::Uscaled(_, _, _) |
583                Command::Uslice(_, _, _) |
584                Command::Urange(_, _, _) |
585                Command::Ulist(_, _) |
586                Command::Ufields(_) |
587                Command::Sbits(_, _) |
588                Command::Sscaled(_, _, _) |
589                Command::Sslice(_, _, _) => (),
590
591                // integer checks don't have anything to check
592                Command::CUbits(_) |
593                Command::CSscaled(_, _) => (),
594
595                _ => panic!("Invalid argument processor")
596            },
597            FlatArg::JumpTarget { ref jump } => match *command {
598                Command::Offset(relocation) => {
599                    // encode the complete relocation. Always starts at the begin of the instruction, and also relative to that
600                    let stmt = jump.clone().encode(4, 4, &[relocation.to_id()]);
601
602                    relocations.push(stmt);
603                },
604                _ => panic!("Invalid argument processor")
605            },
606            FlatArg::Lit { ref ident } => match *command {
607
608                // Condition codes, literals
609                Command::Cond(offset) => {
610                    let name = ident.to_string();
611                    let bits = *COND_MAP.get(&&*name).expect("bad command data");
612                    statics.push((offset, u32::from(bits)))
613                },
614                Command::CondInv(offset) => {
615                    let name = ident.to_string();
616                    let bits = *COND_MAP.get(&&*name).expect("bad command data");
617                    statics.push((offset, u32::from(bits) ^ 1))
618                },
619                Command::LitList(offset, listname) => {
620                    let name = ident.to_string();
621                    let list = SPECIAL_IDENT_MAP.get(listname).expect("bad command data");
622                    if let Some(&bits) = list.get(&&*name) {
623                        statics.push((offset, bits));
624                    } else {
625                        emit_error!(ident, "Unknown literal");
626                        return Err(None);
627                    }
628                },
629                _ => panic!("Invalid argument processor")
630            }
631        }
632
633        // figure out how far the cursor has to be advanced.
634        match *command {
635            Command::Uslice(_, _, _) |
636            Command::Sslice(_, _, _) => (),
637            Command::CUbits(_) |
638            Command::CUsum(_) |
639            Command::CSscaled(_, _) |
640            Command::CUrange(_, _) => (),
641            _ => cursor += 1
642        }
643    }
644
645    // sanity
646    if cursor != data.args.len() {
647        panic!("Not enough command processors");
648    }
649
650    // apply all statics to bits
651    let mut bits = data.data.base;
652    for (offset, value) in statics {
653        bits |= value << offset;
654    }
655
656    // generate code to be emitted for dynamics
657    if !dynamics.is_empty() {
658        let mut res = quote!{
659            #bits
660        };
661        for (offset, expr) in dynamics {
662            res = quote!{
663                #res | ((#expr) << #offset)
664            };
665        }
666        ctx.state.stmts.push(Stmt::ExprUnsigned(delimited(res), Size::B_4));
667    } else {
668        ctx.state.stmts.push(Stmt::Const(u64::from(bits), Size::B_4));
669    }
670
671    // generate code to be emitted for relocations
672    ctx.state.stmts.extend(relocations);
673
674    Ok(())
675}

677fn handle_special_immediates(offset: u8, special: SpecialComm, imm: &syn::Expr, statics: &mut Vec<(u8, u32)>, dynamics: &mut Vec<(u8, TokenStream)>) -> Result<(), Option<String>> {
678    match special {
679        SpecialComm::INVERTED_WIDE_IMMEDIATE_X => if let Some(number) = None::<u64> { // as_unsigned_number(imm) {
680            if let Some(encoded) = encoding_helpers::encode_wide_immediate_64bit(!number) {
681                statics.push((offset, encoded));
682                return Ok(());
683            }
684        } else {
685            dynamics.push((offset, quote_spanned!{ imm.span()=>
686                {
687                    let value: u64 = !#imm;
688                    let offset = value.trailing_zeros() & 0b110000;
689
690                    if (value & !(0xFFFFu64 << offset)) != 0 {
691                        ::dynasmrt::aarch64::immediate_out_of_range_unsigned_64(!value);
692                    }
693
694                    ((0xFFFFu64 & (value >> offset)) as u32) | (offset << 12)
695                }
696            }));
697            return Ok(());
698        },
699        SpecialComm::INVERTED_WIDE_IMMEDIATE_W => if let Some(number) = as_unsigned_number(imm) {
700            if number <= u64::from(u32::MAX) {
701                if let Some(encoded) = encoding_helpers::encode_wide_immediate_32bit(!(number as u32)) {
702                    statics.push((offset, encoded));
703                    return Ok(());
704                }
705            }
706        } else {
707            dynamics.push((offset, quote_spanned!{ imm.span()=>
708                {
709                    let value: u32 = !#imm;
710                    let offset = value.trailing_zeros() & 0b10000;
711
712                    if (value & !(0xFFFFu32 << offset)) != 0 {
713                        ::dynasmrt::aarch64::immediate_out_of_range_unsigned_32(!value);
714                    }
715
716                    (0xFFFFu32 & (value >> offset)) | (offset << 12)
717                }
718            }));
719            return Ok(());
720        },
721        SpecialComm::WIDE_IMMEDIATE_X => if let Some(number) = as_unsigned_number(imm) {
722            if let Some(encoded) = encoding_helpers::encode_wide_immediate_64bit(number) {
723                statics.push((offset, encoded));
724                return Ok(());
725            }
726        } else {
727            dynamics.push((offset, quote_spanned!{ imm.span()=>
728                {
729                    let value: u64 = #imm;
730                    let offset = value.trailing_zeros() & 0b110000;
731
732                    if (value & !(0xFFFFu64 << offset)) != 0 {
733                        ::dynasmrt::aarch64::immediate_out_of_range_unsigned_64(value);
734                    }
735
736                    ((0xFFFFu64 & (value >> offset)) as u32) | (offset << 12)
737                }
738            }));
739            return Ok(());
740        },
741        SpecialComm::WIDE_IMMEDIATE_W => if let Some(number) = as_unsigned_number(imm) {
742            if number <= u64::from(u32::MAX) {
743                if let Some(encoded) = encoding_helpers::encode_wide_immediate_32bit(number as u32) {
744                    statics.push((offset, encoded));
745                    return Ok(());
746                }
747            }
748        } else {
749            dynamics.push((offset, quote_spanned!{ imm.span()=>
750                {
751                    let value: u32 = #imm;
752                    let offset = value.trailing_zeros() & 0b10000;
753
754                    if (value & !(0xFFFFu32 << offset)) != 0 {
755                        ::dynasmrt::aarch64::immediate_out_of_range_unsigned_32(value);
756                    }
757
758                    (0xFFFFu32 & (value >> offset)) | (offset << 12)
759                }
760            }));
761            return Ok(());
762        },
763        SpecialComm::STRETCHED_IMMEDIATE => if let Some(number) = as_unsigned_number(imm) {
764            if let Some(encoded) = encoding_helpers::encode_stretched_immediate(number) {
765                statics.push((offset, encoded & 0x1F));
766                statics.push((offset + 6, encoded & 0xE0));
767                return Ok(());
768            }
769        } else {
770            dynamics.push((offset, quote_spanned!{ imm.span()=>
771                {
772                    let value: u64 = #imm;
773                    let mut test = value & 0x0101_0101_0101_0101;
774                    test |= test << 1;
775                    test |= test << 2;
776                    test |= test << 4;
777                    if test != value {
778                        ::dynasmrt::aarch64::immediate_out_of_range_unsigned_64(value);
779                    }
780                    let mut masked = value & 0x8040201008040201;
781                    masked |= masked >> 32;
782                    masked |= masked >> 16;
783                    masked |= masked >> 8;
784                    let masked = masked as u32;
785                    ((masked & 0xE0) << 6) | (masked & 0x1F)
786                }
787            }));
788            return Ok(());
789        },
790        SpecialComm::LOGICAL_IMMEDIATE_W => if let Some(number) = as_unsigned_number(imm) {
791            if number <= u64::from(u32::MAX) {
792                if let Some(encoded) = encoding_helpers::encode_logical_immediate_32bit(number as u32) {
793                    statics.push((offset, u32::from(encoded)));
794                    return Ok(());
795                }
796            }
797        } else {
798            dynamics.push((offset, quote_spanned!{ imm.span()=>
799                dynasmrt::aarch64::encode_logical_immediate_32bit(#imm).unwrap_or_else(
800                    || ::dynasmrt::aarch64::immediate_out_of_range_unsigned_32(#imm)) as u32
801            }));
802            return Ok(());
803        },
804        SpecialComm::LOGICAL_IMMEDIATE_X => if let Some(number) = as_unsigned_number(imm) {
805            if let Some(encoded) = encoding_helpers::encode_logical_immediate_64bit(number) {
806                statics.push((offset, u32::from(encoded)));
807                return Ok(());
808            }
809        } else {
810            dynamics.push((offset, quote_spanned!{ imm.span()=>
811                dynasmrt::aarch64::encode_logical_immediate_64bit(#imm).unwrap_or_else(
812                    || ::dynasmrt::aarch64::immediate_out_of_range_unsigned_64(#imm)) as u32
813            }));
814            return Ok(());
815        },
816        SpecialComm::FLOAT_IMMEDIATE => if let Some(number) = as_float(imm) {
817            if let Some(encoded) = encoding_helpers::encode_floating_point_immediate(number as f32) {
818                statics.push((offset, u32::from(encoded)));
819                return Ok(());
820            }
821        } else {
822            dynamics.push((offset, quote_spanned!{ imm.span()=>
823                {
824                    let value: f32 = #imm;
825                    let bits = value.to_bits();
826
827                    let check = (bits >> 25) & 0x3F;
828                    if (check != 0b10_0000 && check != 0b01_1111) || (bits & 0x7_FFFF) != 0 {
829                        ::dynasmrt::aarch64::immediate_out_of_range_unsigned_f32(value);
830                    }
831
832                    ((bits >> 24) & 0x80) | ((bits >> 19) & 0x7F)
833                }
834            }));
835            return Ok(());
836        },
837        SpecialComm::SPLIT_FLOAT_IMMEDIATE => if let Some(number) = as_float(imm) {
838            if let Some(encoded) = encoding_helpers::encode_floating_point_immediate(number as f32) {
839                statics.push((offset, u32::from(encoded & 0x1F)));
840                statics.push((offset + 6, u32::from(encoded & 0xE0)));
841                return Ok(());
842            }
843        } else {
844            dynamics.push((offset, quote_spanned!{ imm.span()=>
845                {
846                    let value: f32 = #imm;
847                    let bits = value.to_bits();
848
849                    let check = (bits >> 25) & 0x3F;
850                    if (check != 0b10_0000 && check != 0b01_1111) || (bits & 0x7_FFFF) != 0 {
851                        ::dynasmrt::aarch64::immediate_out_of_range_unsigned_f32(value);
852                    }
853
854                    ((bits >> 18) & 0x20_00) | ((bits >> 13) & 0x18_00) | ((bits >> 19) & 0x1F)
855                }
856            }));
857            return Ok(());
858        },
859    }
860
861    emit_error!(imm, "Unrepresentable immediate");
862    Err(None)
863}

865/// statically checks the value of expr.
866/// the check performed is effectively ((expr >> scale) + bias) <= range
867/// as well as that ((expr >> scale) << scale) == expr
868/// this function operates only on 32-bit numbers, both signed and unsigned
869/// `bias` specifies the lowest value that the expression can have, while `range`
870/// specifies the highest value, after bias has been subtracted.
871/// if the expression cannot be statically evaluated, `Ok(None) is returned`
872/// if it can be evaluated, `Ok(Some(biased, unbiased))` is returned if it meets the range check.
873/// else, an `Err` is returned.
874fn static_range_check(expr: &syn::Expr, bias: i32, range: u32, scale: u8) -> Result<Option<(u32, u32)>, Option<String>> {
875    #![allow(unexpected_cfgs)]
876
877    // signed 64-bit parse is always safe for 32-bit numbers
878    let value = match as_signed_number(expr) {
879        Some(v) => v,
880        None => return Ok(None)
881    };
882
883    // this allows turning off static checks for testing purposes
884    #[cfg(disable_static_checks="1")]
885    return Ok(None);
886
887
888    // arithmetic right shift
889    let scaled: i64 = value >> scale;
890    if scaled << scale != value {
891        emit_error!(expr, "Unrepresentable immediate");
892        return Err(None);
893    }
894
895    let biased = scaled - i64::from(bias);
896    if biased < 0 {
897        emit_error!(expr, "Immediate too low");
898        Err(None)
899    } else if biased > i64::from(range) {
900        emit_error!(expr, "Immediate too high");
901        Err(None)
902    } else {
903        // this cast is always safe
904        Ok(Some((biased as u32, scaled as u32)))
905    }
906}

908/// emits the code for a range check on an unsigned immediate.
909fn dynamic_range_check_unsigned(span: Span, bias: u32, range: u32, scale: u8) -> TokenStream {
910    let check = if scale == 0 {
911        if bias == 0 {
912            quote_spanned!{ span=> _dyn_imm > #range }
913        } else {
914            quote_spanned!{ span=> _dyn_imm.wrapping_sub(#bias) > #range }
915        }
916    } else {
917        let mask = bitmask(scale);
918
919        if bias == 0 {
920            quote_spanned!{ span=> ((_dyn_imm & #mask) != 0) || (_dyn_imm >> #scale) > #range }
921        } else {
922            quote_spanned!{ span=> ((_dyn_imm & #mask) != 0) || (_dyn_imm >> #scale).wrapping_sub(#bias) > #range }
923        }
924    };
925
926    quote_spanned!{ span => if #check { ::dynasmrt::aarch64::immediate_out_of_range_unsigned_32(_dyn_imm); }}
927}

929/// emits the code for a range check on a signed immediate.
930fn dynamic_range_check_signed(span: Span, bias: i32, range: u32, scale: u8) -> TokenStream {
931    let bias = -bias;
932
933    let check = if scale == 0 {
934        if bias == 0 {
935            quote_spanned!{ span => (_dyn_imm as u32) > #range }
936        } else {
937            quote_spanned!{ span => (_dyn_imm.wrapping_add(#bias) as u32) > #range }
938        }
939    } else {
940        let mask = bitmask(scale) as i32;
941
942        if bias == 0 {
943            quote_spanned!{ span=> ((_dyn_imm & #mask) != 0) || ((_dyn_imm >> #scale) as u32) > #range }
944        } else {
945            quote_spanned!{ span=> ((_dyn_imm & #mask) != 0) || ((_dyn_imm >> #scale).wrapping_add(#bias) as u32) > #range }
946        }
947    };
948
949    quote_spanned!{ span => if #check { ::dynasmrt::aarch64::immediate_out_of_range_signed_32(_dyn_imm); }}
