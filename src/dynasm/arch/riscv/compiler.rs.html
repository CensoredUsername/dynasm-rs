13/// Compile a single instruction. Input is taken from `data`, containing both the arguments
14/// and the encoding template and commands.
15/// Output is written to ctx.state
16/// Errors can be emitted to either the whole instruction span (by returning Err(Some(errormsg)))
17/// or emited specifically using emit_error! and returning Err(None)
18pub(super) fn compile_instruction(ctx: &mut Context, data: MatchData) -> Result<(), Option<String>> {
19    // argument cursor
20    let mut cursor = 0usize;
21
22    // All static bitfields (compile-time constant) will be encoded into this map of (offset, bitfield)
23    let mut statics = Vec::new();
24    // All dynamic bitfields (run-time determined) will be encoded into this map of (offset, TokenStream)
25    let mut dynamics = Vec::new();
26    // Any relocation will be encoded in this list
27    let mut relocations = Vec::new();
28
29    for (i, command) in data.data.commands.iter().enumerate() {
30        // meta commands
31        match *command {
32
33            Command::Repeat => {
34                cursor -= 1;
35                continue;
36            },
37            Command::Next => {
38                cursor += 1;
39                continue;
40            },
41            _ => ()
42        }
43
44        let arg = data.args.get(cursor).expect("Invalid encoding data, tried to process more arguments than given");
45
46        match *arg {
47            FlatArg::Register { span, reg: Register::Static(id) } => {
48                let mut code = id.code();
49
50                let offset = match *command {
51                    Command::R(offset) => offset,
52                    Command::Reven(offset) => {
53                        if code & 0x1 != 0 {
54                            emit_error!(span, "This register must be even numbered");
55                            return Err(None);
56                        }
57                        offset
58                    },
59                    Command::Rno0(offset) => {
60                        if code == 0 {
61                            emit_error!(span, "This register must not be x0");
62                            return Err(None);
63                        }
64                        offset
65                    },
66                    Command::Rno02(offset) => {
67                        if code == 0 || code == 2 {
68                            emit_error!(span, "This register must not be x0 or x2");
69                            return Err(None);
70                        }
71                        offset
72                    },
73                    Command::Rpop(offset) => {
74                        if code < 8 || code > 15 {
75                            emit_error!(span, "This register must be one of x8-x15");
76                            return Err(None);
77                        }
78                        code &= 7;
79                        offset
80                    },
81                    Command::Rpops(offset) => {
82                        if ((1 << code) & 0x00_FC_03_00) == 0 {
83                            emit_error!(span, "This register must be one of s0-s7 (x8, x9, x18-x23");
84                            return Err(None);
85                        }
86                        code &= 7;
87                        offset
88                    },
89                    Command::Rpops2(offset) => {
90                        match data.args.get(cursor - 1) {
91                            Some(FlatArg::Register { reg: Register::Static(id2), .. } ) => {
92                                if *id2 == id {
93                                    emit_error!(span, "This register cannot be identical to the previous argument");
94                                    return Err(None);
95                                }
96                            },
97                            Some(FlatArg::Register { reg: Register::Dynamic(_, ref expr), .. }) => {
98                                dynamics.push((0, quote_spanned!{ span=>
99                                    {
100                                        let _dyn_reg: u8 = #expr;
101                                        if _dyn_reg == #code {
102                                            ::dynasmrt::riscv::invalid_register(#code);
103                                        }
104                                        0u32
105                                    }
106                                }));
107                            },
108                            _ => panic!("Invalid encoding data, expected a register before")
109                        }
110
111                        if ((1 << code) & 0x00_FC_03_00) == 0 {
112                            emit_error!(span, "This register must be one of s0-s7 (x8, x9, x18-x23");
113                            return Err(None);
114                        }
115                        code &= 7;
116                        offset
117                    },
118                    _ => panic!("Invalid argument processor")
119                };
120
121                statics.push((offset, u32::from(code)));
122            },
123
124            FlatArg::Register { span, reg: Register::Dynamic(_, ref expr) } => match *command {
125                Command::R(offset) => {
126                    dynamics.push((offset, quote_spanned!{ span=>
127                        ((#expr & 0x1F) as u32)
128                    }));
129                },
130                Command::Reven(offset) => {
131                    let invalid_reg_mask: u8 = if ctx.target.is_embedded() { 0xF0 } else { 0xE0 };
132                    dynamics.push((offset, quote_spanned!{ span=>
133                        {
134                            let _dyn_reg: u8 = #expr;
135                            if _dyn_reg & 0x1 != 0x0 || (_dyn_reg & #invalid_reg_mask) != 0 {
136                                ::dynasmrt::riscv::invalid_register(_dyn_reg);
137                            }
138                            (_dyn_reg & 0x1E) as u32
139                        }
140                    }));
141                },
142                Command::Rno0(offset) => {
143                    let invalid_reg_mask: u8 = if ctx.target.is_embedded() { 0xF0 } else { 0xE0 };
144                    dynamics.push((offset, quote_spanned!{ span=>
145                        {
146                            let _dyn_reg: u8 = #expr;
147                            if _dyn_reg == 0x0 || (_dyn_reg & #invalid_reg_mask) != 0 {
148                                ::dynasmrt::riscv::invalid_register(_dyn_reg);
149                            }
150                            (_dyn_reg & 0x1F) as u32
151                        }
152                    }));
153                },
154                Command::Rno02(offset) => {
155                    let invalid_reg_mask: u8 = if ctx.target.is_embedded() { 0xF0 } else { 0xE0 };
156                    dynamics.push((offset, quote_spanned!{ span=>
157                        {
158                            let _dyn_reg: u8 = #expr;
159                            if _dyn_reg == 0x0 || _dyn_reg == 0x2 || (_dyn_reg & #invalid_reg_mask) != 0 {
160                                ::dynasmrt::riscv::invalid_register(_dyn_reg);
161                            }
162                            (_dyn_reg & 0x1F) as u32
163                        }
164                    }));
165                },
166                Command::Rpop(offset) => {
167                    let invalid_reg_mask: u8 = if ctx.target.is_embedded() { 0xF0 } else { 0xE0 };
168                    dynamics.push((offset, quote_spanned!{ span=>
169                        {
170                            let _dyn_reg: u8 = #expr;
171                            if _dyn_reg & 0x18 != 0x8 || (_dyn_reg & #invalid_reg_mask) != 0 {
172                                ::dynasmrt::riscv::invalid_register(_dyn_reg);
173                            }
174                            (_dyn_reg & 0x7) as u32
175                        }
176                    }));
177                },
178                Command::Rpops(offset) => {
179                    let invalid_reg_mask: u8 = if ctx.target.is_embedded() { 0xF0 } else { 0xE0 };
180                    dynamics.push((offset, quote_spanned!{ span=>
181                        {
182                            let _dyn_reg: u8 = #expr;
183                            if (1u32 << (_dyn_reg & 0x1F)) & 0x00_FC_03_00 == 0 || (_dyn_reg & #invalid_reg_mask) != 0 {
184                                ::dynasmrt::riscv::invalid_register(_dyn_reg);
185                            }
186                            (_dyn_reg & 0x7) as u32
187                        }
188                    }));
189                },
190                Command::Rpops2(offset) => match data.args.get(cursor - 1) {
191                    Some(FlatArg::Register { reg: Register::Static(id2), .. } ) => {
192                        let code: u8 = id2.code();
193                        let invalid_reg_mask: u8 = if ctx.target.is_embedded() { 0xF0 } else { 0xE0 };
194                        dynamics.push((offset, quote_spanned!{ span=>
195                            {
196                                let _dyn_reg: u8 = #expr;
197                                if (_dyn_reg == #code) || ((1u32 << (_dyn_reg & 0x1F)) & 0x00_FC_03_00 == 0) || (_dyn_reg & #invalid_reg_mask) != 0 {
198                                    ::dynasmrt::riscv::invalid_register(_dyn_reg);
199                                }
200                                (_dyn_reg & 0x7) as u32
201                            }
202                        }));
203                    },
204                    Some(FlatArg::Register { reg: Register::Dynamic(_, ref expr2), .. }) => {
205                        let invalid_reg_mask: u8 = if ctx.target.is_embedded() { 0xF0 } else { 0xE0 };
206                        dynamics.push((offset, quote_spanned!{ span=>
207                            {
208                                let _dyn_reg: u8 = #expr;
209                                let _dyn_reg_prev: u8 = #expr2;
210                                if (_dyn_reg == _dyn_reg_prev) || ((1u32 << (_dyn_reg & 0x1F)) & 0x00_FC_03_00 == 0) || (_dyn_reg & #invalid_reg_mask) != 0 {
211                                    ::dynasmrt::riscv::invalid_register(_dyn_reg);
212                                }
213                                (_dyn_reg & 0x7) as u32
214                            }
215                        }));
216                    },
217                    _ => panic!("Invalid encoding data, expected a register before")
218                },
219                _ => panic!("Invalid argument processor")
220            },
221
222            FlatArg::RegisterList { span, ref count } => match *command {
223                Command::Rlist(offset) => match count {
224                    RegListFlat::Static(c) => {
225                        statics.push((offset, u32::from(*c)));
226                    },
227                    RegListFlat::Dynamic(expr) => {
228                        if let Some(static_value) = as_signed_number(expr) {
229                            if static_value < 0 || (static_value > 10 && static_value != 12) {
230                                emit_error!(expr, "Impossible register list");
231                                return Err(None);
232                            }
233
234                            let c = if static_value == 12 {
235                                15
236                            } else {
237                                (static_value + 4) as u32
238                            };
239
240                            statics.push((offset, c));
241
242                        } else {
243                            // okay, so we're given an expression here, with legal input values
244                            // being 0-10 and 12, which should be mapped to 4-14 and 15
245                            // note: this isn't quite the right register error
246                            dynamics.push((offset, quote_spanned!{ span=>
247                                {
248                                    let _dyn_reg: u32 = #expr;
249                                    if _dyn_reg == 11 || _dyn_reg > 12 {
250                                        ::dynasmrt::riscv::invalid_register(_dyn_reg as u8);
251                                    }
252                                    (_dyn_reg + if (_dyn_reg == 12) { 3 } else { 4 }) & 0xF
253                                }
254                            }));
255                        }
256                    }
257                },
258                _ => panic!("Invalid argument processor")
259            },
260
261            FlatArg::Default => match *command {
262                // Default is only emitted for a RefOffset where no offset was provided, i.e. it is 0
263                // This practically means we just don't have to encode anything.
264                Command::UImm(_, _)
265                | Command::SImm(_, _)
266                | Command::BitRange(_, _, _) 
267                | Command::Next => (),
268                _ => panic!("Invalid argument processor")
269            },
270
271            FlatArg::Immediate { ref value } => match *command {
272                // integer verification commands
273                Command::UImm(bits, scaling) => {
274                    let span = value.span();
275                    let range: u32 = bitmask(bits);
276
277                    let mut imm_encoder = ImmediateEncoder::new(value);
278                    imm_encoder.gather_fields(data.data.commands, i + 1, &mut statics);
279
280                    match imm_encoder.static_value {
281                        Some(static_value) => {
282                            static_range_check(static_value, 0, range, scaling, span)?;
283                        },
284                        None => {
285                            let check = if scaling == 0 {
286                                quote_spanned!{ span =>
287                                    _dyn_imm > #range
288                                }
289                            } else {
290                                let zeromask: u32 = bitmask(scaling);
291                                quote_spanned!{ span =>
292                                    _dyn_imm > #range || _dyn_imm & #zeromask != 0u32
293                                }
294                            };
295                            imm_encoder.emit_dynamic(false, false, check, &mut dynamics);
296                        }
297                    }
298                },
299                Command::SImm(bits, scaling) => {
300                    let span = value.span();
301                    let range = bitmask(bits);
302                    let min: i32 = (-1) << (bits - 1);
303
304                    let mut imm_encoder = ImmediateEncoder::new(value);
305                    imm_encoder.gather_fields(data.data.commands, i + 1, &mut statics);
306
307                    match imm_encoder.static_value {
308                        Some(static_value) => {
309                            static_range_check(static_value, min, range, scaling, span)?;
310                        },
311                        None => {
312                            let check = if scaling == 0 {
313                                quote_spanned!{ span =>
314                                    _dyn_imm.wrapping_sub(#min) as u32 > #range
315                                }
316                            } else {
317                                let zeromask = bitmask(scaling) as i32;
318                                quote_spanned!{ span =>
319                                    _dyn_imm.wrapping_sub(#min) as u32 > #range || _dyn_imm & #zeromask != 0i32
320                                }
321                            };
322                            imm_encoder.emit_dynamic(true, false, check, &mut dynamics);
323                        }
324                    }
325                },
326                Command::BigImm(bits) => {
327                    let span = value.span();
328                    let range = bitmask64(bits);
329                    let min: i64 = (-1) << (bits - 1);
330
331                    let mut imm_encoder = ImmediateEncoder::new(value);
332                    imm_encoder.gather_fields(data.data.commands, i + 1, &mut statics);
333
334                    match imm_encoder.static_value {
335                        Some(static_value) => {
336                            if static_value < min {
337                                emit_error!(span, "Immediate too low");
338                                return Err(None);
339                            }
340                            if static_value.wrapping_sub(min) as u64 > range {
341                                emit_error!(span, "Immediate too high");
342                                return Err(None);
343                            }
344                        },
345                        None => {
346                            let check = quote_spanned!{ span =>
347                                _dyn_imm.wrapping_sub(#min) as u64 > #range
348                            };
349                            imm_encoder.emit_dynamic(true, true, check, &mut dynamics);
350                        }
351                    }
352                },
353                Command::UImmNo0(bits, scaling) => {
354                    let span = value.span();
355                    let range = bitmask(bits);
356
357                    let mut imm_encoder = ImmediateEncoder::new(value);
358                    imm_encoder.gather_fields(data.data.commands, i + 1, &mut statics);
359
360                    match imm_encoder.static_value {
361                        Some(static_value) => {
362                            static_range_check(static_value, 0, range, scaling, span)?;
363                            if static_value == 0 {
364                                emit_error!(span, "Immediate cannot be zero");
365                                return Err(None);
366                            }
367                        },
368                        None => {
369                            let check = if scaling == 0 {
370                                quote_spanned!{ span =>
371                                    _dyn_imm > #range || _dyn_imm == 0u32
372                                }
373                            } else {
374                                let zeromask: u32 = bitmask(scaling);
375                                quote_spanned!{ span =>
376                                    _dyn_imm > #range || _dyn_imm & #zeromask != 0u32 || _dyn_imm == 0u32
377                                }
378                            };
379                            imm_encoder.emit_dynamic(false, false, check, &mut dynamics);
380                        },
381                    }
382                },
383                Command::SImmNo0(bits, scaling) => {
384                    let span = value.span();
385                    let range = bitmask(bits);
386                    let min: i32 = (-1) << (bits - 1);
387
388                    let mut imm_encoder = ImmediateEncoder::new(value);
389                    imm_encoder.gather_fields(data.data.commands, i + 1, &mut statics);
390
391                    match imm_encoder.static_value {
392                        Some(static_value) => {
393                            static_range_check(static_value, min, range, scaling, span)?;
394                            if static_value == 0 {
395                                emit_error!(span, "Immediate cannot be zero");
396                                return Err(None);
397                            }
398                        },
399                        None => {
400                            let check = if scaling == 0 {
401                                quote_spanned!{ span =>
402                                    _dyn_imm.wrapping_sub(#min) as u32 > #range || _dyn_imm == 0i32
403                                }
404                            } else {
405                                let zeromask = bitmask(scaling) as i32;
406                                quote_spanned!{ span =>
407                                    _dyn_imm.wrapping_sub(#min) as u32 > #range || _dyn_imm & #zeromask != 0i32 || _dyn_imm == 0i32
408                                }
409                            };
410                            imm_encoder.emit_dynamic(true, false, check, &mut dynamics);
411                        }
412                    }
413                },
414                Command::UImmOdd(bits, scaling) => {
415                    let span = value.span();
416                    let range = bitmask(bits);
417                    let zeromask: u32 = bitmask(scaling);
418
419                    let mut imm_encoder = ImmediateEncoder::new(value);
420                    imm_encoder.gather_fields(data.data.commands, i + 1, &mut statics);
421
422                    match imm_encoder.static_value {
423                        Some(static_value) => {
424                            let biased = static_range_check(static_value, 0, range, 0, span)?;
425                            if biased & zeromask != zeromask {
426                                emit_error!(span, "Unrepresentable immediate");
427                                return Err(None);
428                            }
429                        },
430                        None => {
431                            let check = if scaling == 0 {
432                                quote_spanned!{ span =>
433                                    _dyn_imm > #range
434                                }
435                            } else {
436                                quote_spanned!{ span =>
437                                    _dyn_imm > #range || _dyn_imm & #zeromask != #zeromask
438                                }
439                            };
440                            imm_encoder.emit_dynamic(false, false, check, &mut dynamics);
441                        }
442                    }
443                },
444                Command::UImmRange(min, max) => {
445                    let span = value.span();
446                    let min = u32::from(min);
447                    let max = u32::from(max);
448
449                    let mut imm_encoder = ImmediateEncoder::new(value);
450                    imm_encoder.gather_fields(data.data.commands, i + 1, &mut statics);
451
452                    match imm_encoder.static_value {
453                        Some(static_value) => {
454                            if static_value < i64::from(min) {
455                                emit_error!(span, "Immediate too low");
456                                return Err(None);
457                            }
458                            if static_value > i64::from(max) {
459                                emit_error!(span, "Immediate too high");
460                                return Err(None);
461                            }
462                        },
463                        None => {
464                            let check = quote_spanned!{ span=>
465                                _dyn_imm < #min || _dyn_imm > #max
466                            };
467                            imm_encoder.emit_dynamic(false, false, check, &mut dynamics);
468                        }
469                    }
470                },
471
472                // integer encoding commands
473                Command::BitRange(offset, bits, scaling) => (),
474                Command::RBitRange(offset, bits, scaling) => (),
475
476                // offsets can accept immediates too
477                Command::Offset(relocation_type) => {
478                    let bits;
479                    let scaling;
480                    let commands: &'static [Command];
481
482                    // equivalent bitrange encodings for offsets
483                    match relocation_type {
484                        // 12 bits, 2-bit scaled. Equivalent to
485                        Relocation::B => {
486                            bits = 12;
487                            scaling = 1;
488                            commands = &[
489                                Command::BitRange(31, 1, 12),
490                                Command::BitRange(25, 6, 5),
491                                Command::BitRange(8, 4, 1),
492                                Command::BitRange(7, 1, 11),
493                                Command::Next
494                            ];
495                        },
496                        // 20 bits, 2-bit scaled
497                        Relocation::J => {
498                            bits = 20;
499                            scaling = 1;
500                            commands = &[
501                                Command::BitRange(31, 1, 20),
502                                Command::BitRange(21, 10, 1),
503                                Command::BitRange(20, 1, 11),
504                                Command::BitRange(12, 8, 12),
505                                Command::Next
506                            ];
507                        },
508                        // 9 bits, 2-bit scaled
509                        Relocation::BC => {
510                            bits = 9;
511                            scaling = 1;
512                            commands = &[
513                                Command::BitRange(12, 1, 8),
514                                Command::BitRange(10, 2, 3),
515                                Command::BitRange(5, 2, 6),
516                                Command::BitRange(3, 2, 1),
517                                Command::BitRange(2, 1, 5),
518                                Command::Next
519                            ];
520                        },
521                        // 12 bits, 2-bit scaled
522                        Relocation::JC => {
523                            bits = 12;
524                            scaling = 1;
525                            commands = &[
526                                Command::BitRange(12, 1, 11),
527                                Command::BitRange(11, 1, 4),
528                                Command::BitRange(9, 2, 8),
529                                Command::BitRange(8, 1, 10),
530                                Command::BitRange(7, 1, 6),
531                                Command::BitRange(6, 1, 7),
532                                Command::BitRange(3, 3, 1),
533                                Command::BitRange(2, 1, 5),
534                                Command::Next
535                            ]; // why
536                        },
537                        // 32 bits, 12-bit scaled, offset by 0x800
538                        Relocation::HI20 => {
539                            bits = 32;
540                            scaling = 0;
541                            commands = &[
542                                Command::RBitRange(12, 20, 12),
543                                Command::Next
544                            ];
545                        },
546                        // 12 bits, no scaling
547                        Relocation::LO12 => {
548                            bits = 12;
549                            scaling = 0;
550                            commands = &[
551                                Command::BitRange(20, 12, 0),
552                                Command::Next
553                            ];
554                        },
555                        // 12 bits, no scaling
556                        Relocation::LO12S => {
557                            bits = 12;
558                            scaling = 0;
559                            commands = &[
560                                Command::BitRange(7, 5, 0),
561                                Command::BitRange(25, 7, 5),
562                                Command::Next
563                            ];
564                        },
565                        // 32 bits, no scaling, offset by 0x800
566                        Relocation::SPLIT32 => {
567                            bits = 32;
568                            scaling = 0;
569                            commands = &[
570                                Command::RBitRange(12, 20, 12),
571                                Command::BitRange(20+32, 12, 0),
572                                Command::Next
573                            ];
574                        },
575                        // 32 bits, no scaling, offset by 0x800
576                        Relocation::SPLIT32S => {
577                            bits = 32;
578                            scaling = 0;
579                            commands = &[
580                                Command::RBitRange(12, 20, 12),
581                                Command::BitRange(7+32, 5, 0),
582                                Command::BitRange(25+32, 7, 5),
583                                Command::Next
584                            ];
585                        },
586                        Relocation::LITERAL8
587                        | Relocation::LITERAL16
588                        | Relocation::LITERAL32
589                        | Relocation::LITERAL64 => panic!("Literal relocation in instruction"),
590                    }
591
592                    let span = value.span();
593                    let mut range = bitmask(bits);
594                    let min: i32 = (-1) << (bits - 1);
595
596                    // special case: the 32-bit AUIPC-based offsets don't actually
597                    // range from -0x8000_0000 to 0x7FFF_FFFF on RV64 due to how
598                    // sign extension interacts between them, they range from
599                    // -0x8000_0800 to 0x7FFF_F7FF. But on RV32 they do span
600                    // from -0x8000_0000 to 0x7FFF_FFFF.
601                    // neither of these limits will ever occur in practical code,
602                    // so for sanity's sake we just clamp to between -0x8000_0000 and
603                    // 0x7FFF_F7FF
604                    if bits == 32 {
605                        range = 0xFFFF_F7FF;
606                    }
607
608                    let mut imm_encoder = ImmediateEncoder::new(value);
609                    imm_encoder.gather_fields(commands, 0, &mut statics);
610
611                    match imm_encoder.static_value {
612                        Some(static_value) => {
613                            static_range_check(static_value, min, range, scaling, span)?;
614                        },
615                        None => {
616                            let check = if scaling == 0 {
617                                quote_spanned!{ span =>
618                                    _dyn_imm.wrapping_sub(#min) as u32 > #range
619                                }
620                            } else {
621                                let zeromask = bitmask(scaling) as i32;
622                                quote_spanned!{ span =>
623                                    _dyn_imm.wrapping_sub(#min) as u32 > #range || _dyn_imm & #zeromask != 0i32
624                                }
625                            };
626
627                            imm_encoder.emit_dynamic(true, false, check, &mut dynamics);
628                        }
629                    }
630                },
631
632                // Non-integer immediate commands
633                Command::RoundingMode(offset) => {
634                    let name = as_ident(value).expect("bad command data").to_string();
635                    if let Some(&bits) = ROUNDMODE_MAP.get(&&*name) {
636                        statics.push((offset, u32::from(bits)));
637                    } else {
638                        emit_error!(value, "Unknown literal");
639                        return Err(None);
640                    }
641                },
642                Command::FenceSpec(offset) => {
643                    let name = as_ident(value).expect("bad command data").to_string();
644                    if let Some(&bits) = FENCESPEC_MAP.get(&&*name) {
645                        statics.push((offset, u32::from(bits)));
646                    } else {
647                        emit_error!(value, "Unknown literal");
648                        return Err(None);
649                    }
650                },
651                Command::Csr(offset) => 'csr: {
652                    // Csr is a bit special as it allows both immediates and names
653                    // because we cannot differentiate from those earlier, we handle that here.
654                    if let Some(name) = as_ident(value) {
655                        let name = name.to_string();
656                        if let Some(&bits) = CSR_MAP.get(&&*name) {
657                            statics.push((offset, u32::from(bits)));
658                            break 'csr;
659                        }
660                    }
661
662                    let span = value.span();
663                    let range = 0xFFF;
664                    let commands = &[
665                        Command::BitRange(offset, 12, 0),
666                        Command::Next
667                    ];
668
669                    let mut imm_encoder = ImmediateEncoder::new(value);
670                    imm_encoder.gather_fields(commands, 0, &mut statics);
671
672                    match imm_encoder.static_value {
673                        Some(static_value) => {
674                            static_range_check(static_value, 0, range, 0, span)?;
675                        },
676                        None => {
677                            let check = quote_spanned!{ span =>
678                                _dyn_imm > #range
679                            };
680                            imm_encoder.emit_dynamic(false, false, check, &mut dynamics);
681                        }
682                    }
683                },
684                Command::FloatingPointImmediate(offset) => 'fpimm: {
685                    // this one has several special cases, and is very hard to do at runtime
686                    // firstly, min, inf and nan are not actually values, but idents
687                    // (min depends on what kind of floating point register the instruction targets)
688                    // next to that, we need to parse 29 floating point values
689                    if let Some(name) = as_ident(value) {
690                        let name = name.to_string();
691                        if let Some(&bits) = FP_IMM_IDENT_MAP.get(&&*name) {
692                            statics.push((offset, u32::from(bits)));
693                            break 'fpimm;
694                        }
695                    }
696
697                    if let Some(value) = as_float(value) {
698                        let value = value as f32;
699
700                        if let Some(&(_, bits)) = FP_IMM_VALUE_MAP.iter().find(|(val, bits)| val == &value) {
701                            statics.push((offset, u32::from(bits)));
702                            break 'fpimm;
703                        }
704                    }
705
706                    emit_error!(value, "Invalid floating point immediate.");
707                    return Err(None);
708                },
709                Command::SPImm(offset, negated) => 'spimm: {
710                    // the value we need to encode here depends on the count of registers in the
711                    // register list, and depends on the target architecture as well.
712                    let count = match data.args.get(cursor - 1) {
713                        Some(FlatArg::RegisterList { count, .. }) => count,
714                        _ => panic!("Invalid encoding data, expected a register list before")
715                    };
716
717                    let span = value.span();
718
719                    // either statically encode it, or return an expression for the register list bias
720                    let bias_expr = match count {
721                        RegListFlat::Static(code) => {
722                            let code = if *code == 15 {16} else {*code};
723
724                            let bias = if ctx.target.is_32_bit() {
725                                i32::from(code / 4 * 16)
726                            } else {
727                                i32::from(code / 2 * 16 - 16)
728                            };
729
730                            if let Some(mut static_value) = as_signed_number(value) {
731                                // statically encode everything
732                                if negated {
733                                    static_value = static_value.saturating_neg();
734                                }
735
736                                let bits = static_range_check(static_value, bias, 48, 4, span)? >> 4;
737                                statics.push((offset, bits));
738
739                                break 'spimm;
740
741                            } else {
742                                quote_spanned!{ span=>
743                                    let _reglist_bias: i32 = #bias;
744                                }
745                            }
746                        },
747                        RegListFlat::Dynamic(expr) => {
748                            if let Some(mut static_value) = as_signed_number(value) {
749                                // just some sanity checks at compile time
750                                if negated {
751                                    static_value = static_value.saturating_neg();
752                                }
753                                if ctx.target.is_32_bit() {
754                                    static_range_check(static_value, 16, 96, 4, span)?;
755                                } else {
756                                    static_range_check(static_value, 16, 144, 4, span)?;
757                                }
758                            }
759                            if ctx.target.is_32_bit() {
760                                quote_spanned!{ span=>
761                                    let _reglist_expr: u8 = #expr;
762                                    let _reglist_bias: i32 = (_reglist_expr as i32) / 4 * 16 + 16;
763                                }
764                            } else {
765                                quote_spanned!{ span=>
766                                    let _reglist_expr: u8 = #expr;
767                                    let _reglist_bias: i32 = (_reglist_expr as i32) / 2 * 16 + 16;
768                                }
769                            }
770                        }
771                    };
772
773                    let imm_expr = if negated {
774                        quote_spanned!{ span=>
775                            let _dyn_imm: i32 = -#value;
776                        }
777                    } else {
778                        quote_spanned!{ span=>
779                            let _dyn_imm: i32 = #value;
780                        }
781                    };
782
783                    dynamics.push((offset, quote_spanned!{ span=>
784                        {
785                            #imm_expr
786                            #bias_expr
787                            if (_dyn_imm < _reglist_bias) || ((_dyn_imm - _reglist_bias) > 48) || ((_dyn_imm & 15) != 0) {
788                                ::dynasmrt::riscv::immediate_out_of_range_signed_32(_dyn_imm);
789                            }
790                            (_dyn_imm - _reglist_bias) as u32 >> 4
791                        }
792                    }));
793                },
794                _ => panic!("Invalid argument processor")
795            },
796
797            FlatArg::JumpTarget { ref jump } => match *command {
798                Command::Offset( relocation ) => {
799                    // encode the complete relocation. Always starts at the begin of the instruction(s), and also relative to that
800                    let stmt = jump.clone().encode(relocation.size(), relocation.size(), &[relocation.to_id()]);
801
802                    relocations.push(stmt);
803                },
804                _ => panic!("Invalid argument processor")
805            }
806        }
807
808        // figure out how far the cursor has to be advanced.
809        match *command {
810            Command::UImm(_, _)
811            | Command::SImm(_, _)
812            | Command::BigImm(_)
813            | Command::UImmNo0(_, _)
814            | Command::SImmNo0(_, _)
815            | Command::UImmOdd(_, _)
816            | Command::UImmRange(_, _)
817            | Command::BitRange(_, _, _)
818            | Command::RBitRange(_, _, _) => (),
819            _ => cursor += 1
820        }
821    }
822
823    // sanity
824    if cursor != data.args.len() {
825        panic!("Not enough command processors");
826    }
827
828    let mut templates = [0u32; 8];
829    let mut exprs = [None, None, None, None, None, None, None, None];
830
831    // for convenience sake we operate in 32 bits width, even for compressed instructions
832    match data.data.template {
833        Template::Compressed(val) => templates[0] = u32::from(val),
834        Template::Single(val) => templates[0] = val,
835        Template::Double(val1, val2) => {
836            templates[0] = val1;
837            templates[1] = val2;
838        },
839        Template::Many(values) => {
840            templates[ .. values.len()].copy_from_slice(values);
841        }
842    };
843
844    // apply all statics to templates
845    for (offset, value) in statics {
846        templates[(offset >> 5) as usize] |= value << (offset & 0x1F);
847    }
848
849    // and process all dynamics
850    for (offset, expr) in dynamics {
851        let index = usize::from(offset >> 5);
852        let offset = offset & 0x1F;
853
854        exprs[index] = match exprs[index].take() {
855            Some(prev_expr) => {
856                Some(if offset == 0 {
857                    quote!{ #prev_expr | #expr }
858                } else {
859                    quote!{ #prev_expr | (#expr << #offset) }
860                })
861            },
862            None => {
863                let bits = templates[index];
864                Some(if offset == 0 {
865                    quote!{ #bits | #expr }
866                } else {
867                    quote!{ #bits | (#expr << #offset) }
868                })
869            }
870        }
871    }
872
873    match data.data.template {
874        Template::Compressed(_) => if let Some(d) = exprs[0].take() {
875            let res = quote!{ (#d) as u16 };
876            ctx.state.stmts.push(Stmt::ExprUnsigned(delimited(res), Size::B_2));
877        } else {
878            ctx.state.stmts.push(Stmt::Const(u64::from(templates[0] & 0xFFFF), Size::B_2));
879        },
880        Template::Single(_) => if let Some(d) = exprs[0].take() {
881            ctx.state.stmts.push(Stmt::ExprUnsigned(delimited(d), Size::B_4));
882        } else {
883            ctx.state.stmts.push(Stmt::Const(u64::from(templates[0]), Size::B_4));
884        },
885        Template::Double(_, _) => {
886            for i in 0 .. 2 {
887                if let Some(d) = exprs[i].take() {
888                    ctx.state.stmts.push(Stmt::ExprUnsigned(delimited(d), Size::B_4));
889                } else {
890                    ctx.state.stmts.push(Stmt::Const(u64::from(templates[i]), Size::B_4));
891                }
892            }
893        },
894        Template::Many(c) => {
895            for i in 0 .. c.len() {
896                if let Some(d) = exprs[i].take() {
897                    ctx.state.stmts.push(Stmt::ExprUnsigned(delimited(d), Size::B_4));
898                } else {
899                    ctx.state.stmts.push(Stmt::Const(u64::from(templates[i]), Size::B_4));
900                }
901            }
902        }
903    }
904
905    ctx.state.stmts.extend(relocations);
906
907    Ok(())
908}

911/// Handles the encoding of immediates in a somewhat efficient fashion.
912struct ImmediateEncoder<'a> {
913    pub dynamic_value: &'a syn::Expr,
914    pub static_value: Option<i64>,
915    pub encodes: Vec<(u8, TokenStream)>, // encoding_offset, expression
916    pub span: Span
917}
918
919impl<'a> ImmediateEncoder<'a> {
920    pub fn new(dynamic_value: &syn::Expr) -> ImmediateEncoder {
921        #![allow(unexpected_cfgs)]
922        let static_value;
923
924        // this allows turning off static checks for testing purposes
925        #[cfg(not(disable_static_checks="1"))]
926        {
927            static_value = as_signed_number(dynamic_value);
928        }
929        #[cfg(disable_static_checks="1")]
930        {
931            static_value = None;
932        }
933
934        let span = dynamic_value.span();
935
936        ImmediateEncoder {
937            dynamic_value,
938            static_value,
939            encodes: Vec::new(),
940            span
941        }
942    }
943
944    pub fn gather_fields(&mut self, commands: &[Command], mut index: usize, statics: &mut Vec<(u8, u32)>) {
945        loop {
946            match commands.get(index) {
947                Some(&Command::BitRange(offset, bits, scaling)) => {
948                    let mask = bitmask(bits);
949
950                    if let Some(v) = self.static_value {
951                        let slice = (v >> scaling) as u32 & mask;
952                        statics.push((offset, slice));
953
954                    } else {
955                        self.encodes.push((offset, quote_spanned!{ self.span=>
956                            ((_dyn_imm >> #scaling) as u32 & #mask)
957                        }));
958                    }
959                },
960                Some(&Command::RBitRange(offset, bits, scaling)) => {
961                    let mask = bitmask(bits);
962                    let round_offset: i64 =  1 << (scaling - 1);
963
964                    if let Some(v) = self.static_value {
965                        let slice = (v.wrapping_add(round_offset) >> scaling) as u32 & mask;
966                        statics.push((offset, slice));
967
968                    } else {
969                        // ensure we emit an unsuffixed literal for this so it works with all
970                        // types of number
971                        let round_offset = Literal::i64_unsuffixed(round_offset);
972                        self.encodes.push((offset, quote_spanned!{ self.span=>
973                            ((_dyn_imm.wrapping_add(#round_offset) >> #scaling) as u32 & #mask)
974                        }));
975                    }
976                },
977                Some(Command::Next) => break,
978                Some(_)
979                | None => panic!("Bad encoding data, integer field sequence is not terminated"),
980            }
981            index += 1;
982        }
983    }
984
985    pub fn emit_dynamic(mut self, is_signed: bool, is_64bit: bool, check: TokenStream, dynamics: &mut Vec<(u8, TokenStream)>) {
986        // assemble encoding chunks
987        let mut exprs = [None, None, None, None, None, None, None, None];
988        let dynamic_value = self.dynamic_value;
989        let span = self.span;
990
991        for (offset, expr) in self.encodes.drain(..) {
992            let index = usize::from(offset >> 5);
993            let offset = offset & 0x1F;
994
995            exprs[index] = match exprs[index].take() {
996                Some(prev_expr) => {
997                    let parenthesized = delimited(prev_expr);
998
999                    Some(if offset == 0 {
1000                        let expr = delimited(expr);
1001                        quote!{ #parenthesized | #expr }
1002                    } else {
1003                        quote!{ #parenthesized | (#expr << #offset) }
1004                    })
1005                },
1006                None => {
1007                    Some(if offset == 0 {
1008                        quote!{ #expr }
1009                    } else {
1010                        quote!{ #expr << #offset }
1011                    })
1012                }
1013            }
1014        }
1015
1016        let imm_ty = match (is_64bit, is_signed) {
1017            (false, false) => quote_spanned!{ span=> u32 },
1018            (false, true)  => quote_spanned!{ span=> i32 },
1019            (true, false)  => quote_spanned!{ span=> u64 },
1020            (true, true)   => quote_spanned!{ span=> i64 }
1021        };
1022
1023        let mut first = true;
1024        for (i, expr) in exprs.into_iter().enumerate() {
1025            let encodes = if let Some(encodes) = expr { encodes } else {
1026                continue
1027            };
1028            let offset = (i * 32) as u8;
1029
1030            if first {
1031                first = false;
1032                let error_expr = match (is_64bit, is_signed) {
1033                    (false, false) => quote_spanned!{ span=>
1034                        ::dynasmrt::riscv::immediate_out_of_range_unsigned_32
1035                    },
1036                    (false, true)  => quote_spanned!{ span=>
1037                        ::dynasmrt::riscv::immediate_out_of_range_signed_32
1038                    },
1039                    (true, false)  => quote_spanned!{ span=>
1040                        ::dynasmrt::riscv::immediate_out_of_range_unsigned_64
1041                    },
1042                    (true, true)   => quote_spanned!{ span=>
1043                        ::dynasmrt::riscv::immediate_out_of_range_signed_64
1044                    }
1045                };
1046
1047                dynamics.push((offset, quote_spanned!{ span=>
1048                    {
1049                        let _dyn_imm: #imm_ty = #dynamic_value;
1050
1051                        if #check {
1052                            #error_expr(_dyn_imm);
1053                        }
1054
1055                        #encodes
1056                    }
1057                }));
1058
1059            } else {
1060                dynamics.push((offset, quote_spanned!{ span=>
1061                    {
1062                        let _dyn_imm: #imm_ty = #dynamic_value;
1063                        #encodes
1064                    }
1065                }));
1066            }
1067        }
1068    }
1069}
1070
1071/// Checks the following things
1072/// value >= min
1073/// (value - min) <= range
1074/// ((value - min) & bitmask(scale)) == 0
1075/// returning (value - min) on success.
1076fn static_range_check(value: i64, min: i32, range: u32, scale: u8, span: Span) -> Result<u32, Option<String>> {
1077    if value < i64::from(min) {
1078        emit_error!(span, "Immediate too low");
1079        return Err(None);
1080    }
1081
1082    let biased = value - i64::from(min);
1083
1084    if biased > i64::from(range) {
1085        emit_error!(span, "Immediate too high");
1086        return Err(None);
1087    }
1088
1089    let biased = biased as u32;
1090
1091    if scale != 0 && (biased & bitmask(scale)) != 0 {
1092        emit_error!(span, "Unrepresentable immediate");
1093        return Err(None);
1094    }
1095
1096    Ok(biased)
