13// syntax for a single op: ident ("." ident)* (arg ("," arg)*)? ";"
14pub(super) fn parse_instruction(ctx: &mut Context, input: parse::ParseStream) -> parse::Result<ast::ParsedInstruction> {
15    let span = input.cursor().span();
16
17    // read the full dot-separated op
18    let mut name = parse_ident_or_rust_keyword(input)?.to_string();
19
20    while input.peek(Token![.]) {
21        let _: Token![.] = input.parse()?;
22        name.push('.');
23
24        if input.peek(syn::LitInt) {
25            let number: syn::LitInt = input.parse()?;
26            name.push_str(number.base10_digits());
27        } else {
28            name.push_str(&parse_ident_or_rust_keyword(input)?.to_string());
29        }
30    }
31
32    let mut args = Vec::new();
33
34    // parse 0 or more comma-separated args
35    if !(input.is_empty() || input.peek(Token![;])) {
36        args.push(parse_arg(ctx, input)?);
37
38        while input.peek(Token![,]) {
39            let _: Token![,] = input.parse()?;
40
41            args.push(parse_arg(ctx, input)?);
42        }
43    }
44
45    // let span = span.join(input.cursor().span()); // FIXME can't join spans ATM
46
47    Ok(ast::ParsedInstruction {
48        name,
49        span,
50        args
51    })
52}

55/// tries to parse a full arg definition
56fn parse_arg(ctx: &mut Context, input: parse::ParseStream) -> parse::Result<ast::RawArg> {
57    let start = input.cursor().span(); // FIXME can't join spans yet
58
59    // a label, identified by a leading < / > / -> / =>
60    if let Some(jump) = input.parse_opt()? {
61        return Ok(ast::RawArg::JumpTarget {
62            jump
63        });
64    }
65
66    // a register list. 
67    //    '{' reg [ ',' reg [ '-' reg ]] '}' 
68    // or '{' reg ';' amount '}'
69    if input.peek(syn::token::Brace) {
70        let span = input.cursor().span();
71        let inner;
72        let _ = syn::braced!(inner in input);
73        let inner = &inner;
74
75        let first = parse_reg(ctx, inner)?.ok_or_else(|| inner.error("Expected register"))?;
76
77        let count = if inner.peek(Token![,]) {
78            let _: Token![,] = inner.parse()?;
79
80            let second = parse_reg(ctx, inner)?.ok_or_else(|| inner.error("Expected register"))?;
81
82            if inner.peek(Token![-]) {
83                let _: Token![-] = inner.parse()?;
84
85                let third = parse_reg(ctx, inner)?.ok_or_else(|| inner.error("Expected register"))?;
86
87                ast::RegListCount::Double(second, third)
88            } else {
89                ast::RegListCount::Single(second)
90            }
91        } else if inner.peek(Token![;]) {
92            let _: Token![;] = inner.parse()?;
93
94            let expr: syn::Expr = inner.parse()?;
95
96            ast::RegListCount::Dynamic(expr)
97        } else {
98            ast::RegListCount::Static(4)
99        };
100
101        return Ok(ast::RawArg::RegisterList {
102            span,
103            first,
104            count
105        });
106    }
107
108    // a memory reference. we use '[' base [ ',' offset ] ']' syntax instead of the natural AT&T
109    // inspired syntax, as this would require large amounts of lookahead in the parser due to the
110    // possible syntactic ambiguities
111    if input.peek(syn::token::Bracket) {
112        let span = input.cursor().span();
113        let inner;
114        let _ = syn::bracketed!(inner in input);
115        let inner = &inner;
116
117        let base = parse_reg(ctx, inner)?.ok_or_else(|| inner.error("Expected register"))?;
118
119        if inner.peek(Token![,]) {
120            let _: Token![,] = inner.parse()?;
121
122            if let Some(jump) = inner.parse_opt()? {
123                return Ok(ast::RawArg::LabelReference {
124                    span,
125                    base,
126                    jump
127                });
128
129            } else {
130
131                let expr: syn::Expr = inner.parse()?;
132
133                return Ok(ast::RawArg::Reference {
134                    span,
135                    base,
136                    offset: Some(expr)
137                });
138            }
139
140        } else {
141            return Ok(ast::RawArg::Reference {
142                span,
143                base,
144                offset: None
145            });
146        };
147    }
148
149    // a register
150    if let Some(reg) = parse_reg(ctx, input)? {
151        return Ok(ast::RawArg::Register {
152            reg,
153            span: start
154        })
155    }
156
157    // immediate
158    // guard against "dyn" used for rounding modes
159    let span = input.span();
160    let expr: syn::Expr = if eat_pseudo_keyword(input, "dyn") {
161        let path: syn::Path = syn::Ident::new("dyn", span).into();
162        syn::ExprPath { attrs: Vec::new(), qself: None, path}.into()
163    } else {
164        input.parse()?
165    };
166
167    Ok(ast::RawArg::Immediate { value: expr })
168}

170/// Checks if the given expression could be a valid RISC-V register reference
171/// This can be a simple register name (like `x5`)
172/// an alias (any simple name that is registered, like `base`)
173/// or a dynamic register (like `X(expr)`)
174/// this is needed because the notation RISC-V uses for memory references is ambiguous, so we
175/// can't distinguish easily at parse time.
176fn parse_reg_from_expression(ctx: &mut Context, expr: &syn::Expr) -> parse::Result<Option<ast::Register>> {
177    Ok(match expr {
178        syn::Expr::Call(exprcall) => {
179            let name = match &*exprcall.func {
180                syn::Expr::Path(exprpath) => match exprpath.path.get_ident() {
181                    Some(ident) => ident.to_string(),
182                    None => return Ok(None)
183                }
184                _ => return Ok(None)
185            };
186
187            if exprcall.args.len() != 1 {
188                return Err(parse::Error::new(
189                    expr.span(),
190                    "Too many arguments in register family expression"
191                ));
192            }
193
194            if let Some(&family) = RISCV_FAMILIES.get(&*name) {
195                Some(ast::Register::Dynamic(family, exprcall.args[0].clone()))
196            } else {
197                None
198            }
199        },
200        syn::Expr::Path(exprpath) => {
201            let mut name = match exprpath.path.get_ident() {
202                Some(ident) => ident.to_string(),
203                None => return Ok(None)
204            };
205
206            // fail if it is a family reference without call expression
207            if RISCV_FAMILIES.contains_key(&*name) {
208                return Err(parse::Error::new(
209                    exprpath.path.span(),
210                    "Register family reference without dynamic register id"
211                ));
212            }
213
214            // check if it is an alias
215            if let Some(repl) = ctx.state.invocation_context.aliases.get(&name) {
216                name = repl.clone();
217            }
218
219            // resolve normal register references
220            RISCV_REGISTERS.get(&*name).cloned().map(ast::Register::Static)
221        },
222        _ =>  None
223    })
224}

226/// Parses a single register, if present
227/// This can be a simple register name (like `x5`)
228/// an alias (any simple name that is registered, like `base`)
229/// or a dynamic register (like `X(expr)`)
230fn parse_reg(ctx: &mut Context, input: parse::ParseStream) -> parse::Result<Option<ast::Register>> {
231    // we need to consume an ident, but only if it's one of the many we care about
232    // so use a step parser to figure it out.
233    let name = input.step(|cursor| {
234        if let Some((ident, rest)) = cursor.ident() {
235            let mut ident = ident.to_string();
236
237            // first, parse known register families
238            if RISCV_FAMILIES.contains_key(&*ident) {
239                return Ok((ident, rest));
240            }
241
242            // otherwise, see if this is an alias
243            if let Some(repl) = ctx.state.invocation_context.aliases.get(&ident) {
244                ident = repl.clone();
245            }
246
247            // resolve normal register references
248            if RISCV_REGISTERS.contains_key(&*ident) {
249                return Ok((ident, rest));
250            }
251        }
252        Err(cursor.error("expected identifier"))
253    });
254
255    let name = match name {
256        Ok(name) => name,
257        Err(_) => return Ok(None)
258    };
259
260    // we know we have a register reference now, try to resolve it.
261    let register = if let Some(&id) = RISCV_REGISTERS.get(&*name) {
262        ast::Register::Static(id)
263    } else if let Some(&family) = RISCV_FAMILIES.get(&*name) {
264        // need to parse the trailing `( expr )`
265        let inner;
266        let _ = syn::parenthesized!(inner in input);
267        let inner = &inner;
268
269        let expr: syn::Expr = inner.parse()?;
270
271        ast::Register::Dynamic(family, expr)
272    } else {
273        unreachable!()
274    };
275
276    Ok(Some(register))
277}
